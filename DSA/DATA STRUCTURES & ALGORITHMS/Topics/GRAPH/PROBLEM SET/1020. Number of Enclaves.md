     int dx[] = {-1, 0, 1, 0}; // Row offsets: up, right, down, left
     int dy[] = {0, 1, 0, -1}; // Col offsets: up, right, down, left

    while(!q.empty()){

        }
       }

    int numEnclaves(vector<vector<int>>& grid) {
