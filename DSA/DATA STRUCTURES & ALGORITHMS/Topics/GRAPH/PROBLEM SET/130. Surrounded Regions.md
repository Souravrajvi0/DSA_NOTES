    for(int i=0;i<oc.size();i++){
        q.push(oc[i]);

     int dx[] = {-1, 0, 1, 0}; // Row offsets: up, right, down, left
     int dy[] = {0, 1, 0, -1}; // Col offsets: up, right, down, left

    while(!q.empty()){
       int size = q.size();

       for(int i=0;i<size;i++){
        pair<int,int>temp=q.front();

                vis[nx][ny]=1;
            }
        }
       }

    void solve(vector<vector<char>>& board) {
        vector<pair<int,int>>oc;
        int n = board.size();

       vector<vector<int>>vis(n,vector<int>(m,0));
       for(int j = 0;j < m;j++){
        if(board[0][j]=='O'){
          oc.push_back({0,j});
          vis[0][j]=1;
        }
        if(board[n-1][j]=='O'){
          oc.push_back({n-1,j});
             vis[n-1][j]=1;
        }
       }
        for(int j = 0;j < n;j++){
        if(board[j][0]=='O'){
          oc.push_back({j,0});
            vis[j][0]=1;

          oc.push_back({j,m-1});
          vis[j][m-1]=1;
        }
       }

       for(int i =0;i<n;i++){
